set(LIBRARY_SRCS

    # Generic int main() file 
    "${CMAKE_CURRENT_SOURCE_DIR}/Src/NSCA.cpp"

)

    "${CMAKE_CURRENT_SOURCE_DIR}/Include/NSCA.h"

  PUBLIC_INCLUDES  ${CMAKE_CURRENT_SOURCE_DIR}/Include
  PRIVATE_INCLUDES ${CMAKE_CURRENT_SOURCE_DIR}/Src/Internal
)
